\begin{frame}{Auto convolução}
\only<1>{\begin{equation}
	f(x) \ast g(x) = \int_{-\infty}^{\infty}f(\tau)g(x-\tau)d\tau \nonumber
	\end{equation}}
\only<2->{\begin{equation}
	rect(t) \ast rect(t) = \int_{-\infty}^{\infty}rect(\tau)rect(t-\tau)d\tau \nonumber
	\end{equation}}
\visible<3->{
\begin{itemize}
	\item Serão feitas 5 auto convoluções;
	\item Por se tratar de convolução, o tamanho de amostragens dobra após cada convolução;
	\item No final, serão comparados os resultados normalizados.
\end{itemize}}
\end{frame}

\begin{frame}{Amplitude espectral}
